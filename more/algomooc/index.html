var intToRoman = function(num) {
  const hashMap = [
    [1, "I"],
    [4, "IV"],
    [5, "V"],
    [9, "IX"],
    [10, "X"],
    [40, "XL"],
    [50, "L"],
    [90, "XC"],
    [100, "C"],
    [400, "CD"],
    [500, "D"],
    [900, "CM"],
    [1000, "M"],
  ];
  let res = "";
  const len = hashMap.length - 1;
  for (let i = len; i >= 0; i--) {
    const key = hashMap[i][0];
    let str = hashMap[i][1];
    const value = Math.floor(num / key);
    const _val = new Array(value).fill(str).join("");
    if (num % key !== 0) {
      res += _val;
      num %= key;
    } else {
      res += _val;
      break;
    }
  }
  return res;
};

var lengthOfLongestSubstring = function(s) {
  if (!s) {
      return 0;
   }
   if (s.length === 1) {
      return 1;
   }
  let str = s[0];
  let res = s[0]
  let i = 0;
  let j = 1;
  while(j <= s.length -1) {
    const code = s[j];
    if (!str.includes(code)) {
      str += code;
      j++;
    } else {
      res = str.length > res.length ? str : res;
      str = '';
      i++;
      j = i;
      continue;
    }
  }
  return Math.max(res.length, str.length)
};
